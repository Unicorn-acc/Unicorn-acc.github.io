class Solution {
    int mod = (int)1e9+7;
    public int minAbsoluteSumDiff(int[] nums1, int[] nums2) {
        int n = nums1.length;
        int[] sorted = nums1.clone();
        Arrays.sort(sorted);
        long max = 0,sum = 0;
        for(int i = 0; i < n; i++){
            int a = nums1[i],b = nums2[i];
            if(a == b) continue;
            int x = Math.abs(a-b);
            sum += x;
            //äºŒåˆ†æŸ¥æ‰¾ï¼Œæ‰¾åˆ°ç¬¬ä¸€ä¸ªå°äºç­‰äºtargetçš„å€¼
            int left = 0,right = n-1;
            while(left <= right){ //[left,right]
                int mid = left + (right-left)/2;
                if(sorted[mid] <= b) left = mid+1;
                else right = mid-1;
            }//æœ€åè¿”å›å€¼ r æˆ–è€…ä¸å­˜åœ¨-1
            int nd = Integer.MAX_VALUE;
            if(right != -1){
                nd = Math.abs(sorted[right] - b);
            }
            if(right + 1 < n) nd = Math.min(nd,Math.abs(sorted[right+1]-b));
            if(nd < x) max = Math.max(max,x-nd);
        }
        return (int) ((sum-max)%mod);
    }
}
