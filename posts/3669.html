class Solution {
    public int countConsistentStrings(String allowed, String[] words) {
        int[] cnt = new int[26];
        for(char c : allowed.toCharArray()){
            cnt[(int)c-'a'] = 1;
        }
        int res = 0;
        for(String word : words){
            boolean issub = true;
            for(char c : word.toCharArray()){
                if(cnt[(int)c-'a'] == 1) continue;
                else{
                    issub = false;
                    break;
                }
            }
            if(issub) res++;
        }
        return res;
    }
}
