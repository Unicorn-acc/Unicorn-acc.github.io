class Solution {
    public String customSortString(String order, String s) {
        Map<Character,Integer> map = new HashMap<>();
        for(int i = 0; i < order.length(); i++){
            map.put(order.charAt(i),i);
        }
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            if(map.containsKey(c)){
                int k = 0;
                int sort = map.get(c);
                String tmp = sb.toString();
                for(;k < sb.toString().length();k++){
                    int st = map.getOrDefault(tmp.charAt(k),-1);
                    if(st != -1 && st < sort) continue;
                    else break;
                }
                sb.insert(k,c);
            }else{
                sb.append(c);
            }
        }
        return sb.toString();
    }
}

class Solution:
    def customSortString(self, order: str, s: str) -> str:
        cnt = Counter(s) 
        # print(cnt) # Counter({'a': 1, 'b': 1, 'c': 1, 'd': 1})
        ans = []
        for c in order:
            ans.append(c * cnt[c])
            cnt[c] = 0
        for c,v in cnt.items():
            ans.append(c * v)
        return ''.join(ans)
