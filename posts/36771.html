class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        TreeSet<Integer> set = new TreeSet<>();
        Map<Integer,Integer> map = new HashMap<>();
        int ans[] = new int[n];
        for(int i = 0; i < n; i++){
            if(rains[i] == 0){
                set.add(i);
                ans[i] = 1;
            }else{
                ans[i] = -1;
                if(map.containsKey(rains[i])){
                    Integer a = set.higher(map.get(rains[i]));
                    if(a == null) return new int[0];
                    else{
                        set.remove(a);
                        ans[a] = rains[i];
                    }
                }
                map.put(rains[i],i);
            }
        }
        return ans;

    }
}

class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        List<Integer> list = new ArrayList<>();
        Map<Integer,Integer> map = new HashMap<>();
        int ans[] = new int[n];
        for(int i = 0; i < n; i++){
            if(rains[i] == 0){
                list.add(i);
                ans[i] = 1;
            }else{
                ans[i] = -1;
                if(map.containsKey(rains[i])){
                    int a = map.get(rains[i]);
