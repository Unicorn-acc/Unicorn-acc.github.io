class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int left = 0,right = (int)1e6;
        int ans = 0;
        while(left < right){
            int mid = left + (right-left)/2;
            int res = check(mid,nums);
            if(res > threshold) {
                left = mid+1;
            }else{
                ans = res;
                right = mid;
            }
        }
        return ans;
    }

    public int check(int res,int[] nums){
        int cnt = 0;
        for(int num : nums){
            cnt += num/res;
        }
        return cnt;
    }
}
