class Solution {
    public boolean isIdealPermutation(int[] nums) {
        int n = nums.length;
        int count1 = 0;
        int[] count = new int[n+1];
        long count2 = 0;
        update(count,nums[0]+1);
        for(int i = 1; i < n; i++){
            if(nums[i] < nums[i-1]){
                count1++;
            }
            count2 += i - get(count,nums[i]);
            update(count,nums[i]+1);
        }
        return count1 == count2;
    }

    void update(int arr[], int k){
        while(k < arr.length){
            arr[k]++;
            k += k&(-k);
        }
    }
    int get(int arr[],int k){
        int ans = 0;
        while(k > 0){
            ans += arr[k];
            k -= k &(-k);
        }
        return ans;
    }
}

class Solution {
    int n, p, g;
    public boolean isIdealPermutation(int[] nums) {
        n = nums.length;
        for (int i = 0; i < n - 1; i++) if (nums[i] > nums[i + 1]) p++;
        merge(nums, 0, n - 1); 
        return g == p;
    }
    void merge(int[] nums, int l, int r) {
        if (l >= r) return;
        int mid = (l + r) >> 1;
        merge(nums, l, mid);
        merge(nums, mid + 1, r);
        int[] tem = new int[r - l + 1];
        int i = l, j = mid + 1, k = 0;
        while (i <= mid && j <= r) {
            if (nums[i] <= nums[j]) tem[k++] = nums[i++];
            else {
                g += mid - i + 1;
                tem[k++] = nums[j++];
            }
        }
        while (i <= mid) tem[k++] = nums[i++];
        while (j <= r) tem[k++] = nums[j++];
        for (i = l, k = 0; i <= r; i++, k++) nums[i] = tem[k];
    }
}
