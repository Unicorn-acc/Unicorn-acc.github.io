class Solution {
    public boolean halvesAreAlike(String s) {
        int n = s.length();
        char[] arr = s.toCharArray();
        int cnt = 0;
        for(int i = 0; i < n/2; i++){
            char c = arr[i];
            if(Isvowel(c)){
                cnt++;
            }
        }
        for(int i = n-1; i >= n/2; i--){
            char c = arr[i];
            if(Isvowel(c)){
                cnt--;
            }
        }
        return cnt == 0 ? true : false;
    }
    public boolean Isvowel(char c){
        if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' ||
            c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U')
            return true;
        else return false;
    }
}

class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        se = set(['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'])
        n = len(s)//2
        return sum(i in se for i in s[:n]) == sum(i in se for i in s[n:])
