class Solution {

    public static final int MOD = (int) (1e9 + 7);

    public int waysToSplit(int[] nums) {
        int n = nums.length;
        // å‰ç¼€å’Œ
        int[] p = new int[n + 1];
        for (int i = 0, t = 0; i < n; i++) {
            t += nums[i];
            p[i + 1] = t;
        }
        long res = 0;
        int tot = p[n];
        for (int i = 1; i < n - 1 && p[i] * 3 <= tot; i++) {
            int left = p[i];
            // äºŒåˆ†æœç´¢æ±‚ä¸‹é™
            int l = i + 1, r = n - 1;
            while (l <= r) {
                int mid = l + r >> 1;
                if (p[mid] - left < left) l = mid + 1;
                else r = mid - 1;
            }
            int j = l;
            // äºŒåˆ†æœç´¢æ±‚ä¸Šé™

                int mid = l + r >> 1;

                else r = mid - 1;
            }
            int k = r;
            // ç´¯åŠ åˆ°ç»“æœä¸­
            res += k - j + 1;
        }
        return (int) (res % MOD);
    }
