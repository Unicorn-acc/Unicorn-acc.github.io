class Solution {
    public int maxProfitAssignment(int[] difficulty, int[] profit, int[] worker) {
        int[][] sorted = new int[difficulty.length][2];
        for(int i = 0; i < difficulty.length; i++){
            sorted[i][0] = difficulty[i];
            sorted[i][1] = profit[i];
        }
        Arrays.sort(sorted,(a,b)-> a[0]-b[0]);
        int max = 0;
        for(int i = 0; i < difficulty.length; i++){
            max = Math.max(max,sorted[i][1]);
            sorted[i][1] = max;
        }
        int res = 0;
        for(int i = 0; i < worker.length; i++){
            int man = worker[i];        
            int left = 0,right = difficulty.length-1;
            while(left <= right){
                //äºŒåˆ†æ‰¾åˆ°ç¬¬ä¸€ä¸ªå°äºç­‰äºmançš„å€¼
                int mid = (left + right)/2;
                if(sorted[mid][0] <= man) left = mid+1;
                else right = mid-1;
            }
            if(right == -1) continue;
            res += sorted[right][1];    
        }
        return res;
    }
}
