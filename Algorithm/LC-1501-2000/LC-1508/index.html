class Solution {
    private static final int MOD = (int)1e9+7;
    public int rangeSum(int[] nums, int n, int left, int right) {
        int[] sum = new int[n*(n+1)/2];
        for(int i = 0,k = 0; i < n; i++){
            int s = 0;
            for(int j = i; j < n; j++){
                s += nums[j];
                sum[k++] = s;
            }
        }
        Arrays.sort(sum);
        int ans = 0;
        for(int i = left-1; i < right; i++){
            ans = (ans + sum[i]) % MOD;
        }
        return ans;
    }
}
