class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        int m = matrix.length,n = matrix[0].length;
        for(int i = 0; i < m; i++){
            int l = 0,r = n-1;
            while(l <= r){
                int mid = l + r >> 1;   
                if(matrix[i][mid] <= target) l = mid+1;
                else r = mid-1;
            }
            if(r == -1) continue;
            if(matrix[i][r] == target) return true;
        }
        return false;
    }
}
