class Solution {
    public int nthUglyNumber(int n, int a, int b, int c) {
        long lcm_ab = lcm(a, b);
        long lcm_bc = lcm(b, c);
        long lcm_ac = lcm(a, c);
        long lcm_abc = lcm(a, lcm(b, c));
        int min = min(a, min(b, c));
        long l = 1, r = min * 1L * n ;
        while (l  < r) {
            long m = l + (r - l) / 2;
            long count = m / a + m / b + m / c - m / lcm_ab - m / lcm_ac - m / lcm_bc + m / lcm_abc;
            if (count < n) {
                l = m+1;
            } else {
                r = m;
            }
        }
        return (int) r;
    }

        return Math.min(a, b);
    }
