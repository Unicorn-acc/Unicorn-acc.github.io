class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        if (m * k > bloomDay.length) {
            return -1;
        }
        int left = 0,right = Arrays.stream(bloomDay).max().getAsInt();
        while(left < right){
            int mid = left + (right-left)/2;
            int cnt = check(mid,bloomDay,k);
            if(cnt >= m) right = mid;
            else{
                left = mid + 1;
            }
        }
        return check(left,bloomDay,k) >= m ? left : -1;
    }
    public int check(int day,int[] nums,int k){
        int count = 0,res = 0;
        for(int i = 0; i < nums.length; i++){
            if(nums[i] <= day){
                count++;
            }else{
                count = 0;
            }
            if(count == k){
                count = 0;
                res++;
            }
        }
        return res;
    }
}
